## 深拷贝

```
function deepClone(obj) {
  if (typeof obj !== 'object' || typeof obj === 'null') return obj;
  let newObj = obj instanceof Array  ? [] : {}; // 重点
  for(let key in obj) {
    if (obj.hasOwnProperty(key)) {
      newObj[key] = typeof obj[key] === 'object' ? deepClone(obj[key]) : obj[key];
    }
  }
  return newObj;
}
```
